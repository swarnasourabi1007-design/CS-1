#include <stdio.h>

int main() {
    int n, f = 0;
    printf("Enter a number: ");
    scanf("%d", &n);

    if (n <= 1 || n % 2 == 0 || n % 3 == 0) {
        if (n == 2 || n == 3)
            f = 0;
        else
            f = 1;
    } else {
        for (int i = 5; i * i <= n; i += 6) {
            if (n % i == 0 || n % (i + 2) == 0) {
                f = 1;
                break;
            }
        }
    }

    if (f == 0)
        printf("%d is a Prime number\n", n);
    else
        printf("%d is Not a Prime number\n", n);

    return 0;
}



Output:
Enter a number: 29
29 is a Prime number
