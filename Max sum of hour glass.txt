#include <stdio.h>
#include <limits.h>
int main() {
    int a[10][10], i, j, sum = 0;
    int max = INT_MIN;
    for (int m = 0; m < 6; m++) {
        for (int n = 0; n < 6; n++) {
            scanf("%d", &a[m][n]);
        }
    }

    for (i = 0; i < 4; i++) {
        for (j = 0; j < 4; j++) {
            sum = a[i][j] + a[i][j+1] + a[i][j+2]
                            + a[i+1][j+1]
                  + a[i+2][j] + a[i+2][j+1] + a[i+2][j+2];
            if (sum > max)
                max = sum;
        }
    }

    printf("%d", max);
    return 0;
}


Output:
19
