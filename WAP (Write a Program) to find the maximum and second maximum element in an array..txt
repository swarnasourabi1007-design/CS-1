#include <stdio.h>

int main() {
    int m1, m2, n;
    printf("Enter size of array: ");
    scanf("%d", &n);

    int a[n];
    printf("Enter %d elements: ", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    // Initialize m1 and m2
    if (a[0] > a[1]) {
        m1 = a[0];
        m2 = a[1];
    } else {
        m1 = a[1];
        m2 = a[0];
    }

    // Loop to find largest and second largest
    for (int i = 2; i < n; i++) {
        if (m1 < a[i]) {
            m2 = m1;
            m1 = a[i];
        } else if (m2 < a[i]) {
            m2 = a[i];
        }
    }

    printf("Maximum = %d\nSecond Maximum = %d\n", m1, m2);

    return 0;
}


Input
Enter size of array: 6
Enter 6 elements: 10 45 32 67 12 50

Output
Maximum = 67
Second Maximum = 50
