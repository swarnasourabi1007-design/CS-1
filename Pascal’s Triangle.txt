#include <stdio.h>

int fact(int n) {
    int p = 1;
    for (int i = 2; i <= n; i++) {
        p *= i;
    }
    return p;
}

int main() {
    int n = 5;   // Example: number of rows

    for (int i = 0; i < n; i++) {
        for (int j = 0; j <= i; j++) {
            printf("%d ", fact(i) / (fact(j) * fact(i - j)));
        }
        printf("\n");
    }
    return 0;
}
