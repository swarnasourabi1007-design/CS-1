#include <bits/stdc++.h>
using namespace std;

int main() {
    int a[] = { -5, 2, -4, 10 };
    int n = sizeof(a)/sizeof(a[0]);
    int sum = 0, c = 0;
    int mn = INT_MAX;

    for (int i = 0; i < n; i++) {
        if (a[i] < 0) c++;  // count negatives
        sum += abs(a[i]);   // take absolute values
        mn = min(mn, abs(a[i]));
    }

    if (c % 2 == 0) {
        cout << sum << endl;
    } else {
        cout << sum - 2 * mn << endl;
    }
    
    return 0;
}
