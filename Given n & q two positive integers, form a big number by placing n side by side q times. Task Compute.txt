#include <stdio.h>

int main() {
    int n, q;
    scanf("%d %d", &n, &q);
    int sum = 0;

    if ((n % 9 == 0 || q % 9 == 0)) {
        sum = q;
    } else {
        int s1 = (n * q) % 9;
        if (s1 % 9 == 0)
            sum = 9;
        else
            sum = (s1 % 9);
    }

    printf("%d", sum);
    return 0;
}
