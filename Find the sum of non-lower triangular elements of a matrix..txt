#include <stdio.h>

int main() {
    int r = 3, c = 3, sum = 0;
    int a[3][3] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9}
    };

    // Traverse the matrix
    for (int i = 0; i < r; i++) {
        for (int j = 0; j < c; j++) {
            if (j >= i) {   // condition for non-lower triangular elements
                sum += a[i][j];
            }
        }
    }

    printf("Sum of non-lower triangular elements = %d\n", sum);
    return 0;
}


Output:
1 + 2 + 3 + 5 + 6 + 9 = 26
